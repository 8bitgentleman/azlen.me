    Return all files of a specific extension from a directory as a list

    def listFilesOfSpecificFormat(directory, extension):

    '''list all files of a specific extension from a directory'''
    specific_files = []

    for file in os.listdir(directory):
    	if file.endswith("." + extension):
    		specific_files.append(os.path.join(directory, file))
    specific_files = sorted(specific_files)
    return specific_files

  • Flatten irregular nested list

    import collections

def flatten(l):
    for el in l:
        if isinstance(el, collections.Iterable) and not isinstance(el, (str, bytes)):
            yield from flatten(el)
        else:
            yield el
